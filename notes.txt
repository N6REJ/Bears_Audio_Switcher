Example of macro that switches between 2 sources.
/run
local x = tonumber(GetCVar("Sound_OutputDriverIndex"));
if x == 0 then SetCVar("Sound_OutputDriverIndex", "3")
 else SetCVar("Sound_OutputDriverIndex", "0");
 end;
 AudioOptionsFrame_AudioRestart();

 Change the two 0's and the 3 to the devices you want to switch between
